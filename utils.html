$.config

설정된 application configuration 정보 get/set.

configuration 정보중 editable:true 인 값들만 runtime 에서 1회 설정이 가능하다.
설정된 정보를 get : readonly. 
								var _info = {
										'useLog' : true
									};
								
								// $.config > set
								$.config(_info);
							
								// $.config > get
								var root = $.config('root');

$.log

$.config('useLog') 설정이 true 이면 로그가 나타난다. 
								$.log('Hello World!');

$.cookie

기본적으로 cookie 는 browser, domain 기준으로 저장되며 string type 을 가진다.
일반적으로 4KB 까지만 저장이 가능합니다.
document.cookie 를 사용할 수 있는 get, set, remove, clear 을 지원한다. 
								// @method get
								// @params name
								$.cookie.get('name');
								
								// @method set
								// @params name
								// @params value
								// @params expireDays {Number}
								$.cookie.set('name', 1);
								
								// @method remove
								// @params name
								$.cookie.remove('name');
								
								// @method clear
								$.cookie.clear();

$.storage

supported IE 8+
localStorage 를 사용하기 위한 interface.
localStorage 는 일반적으로 4MB의 용량을 지원하며 String 으로 저장된다.
								// @method get
								// @params name
								$.storage.get('name');
								
								// @method set
								// @params name
								// @params value
								$.storage.set('name', 1);
								
								// @method remove
								// @params name
								$.storage.remove('name');
								
								// @method clear
								$.storage.clear();

$.stringify

JSON 객체를 지원하지 않는 IE 7 이하 브라우저에서 사용.
JSON.parse(), JSON.stringify() method 형태
$.parseJSON(), $.stringify() 로 대체 사용 가능. 
								var data = {"nation": "korea", "name": "jeros"};
								
								data = JSON.stringify(data);
								$.log($.type(data));					// string
								
								data = JSON.parse(data);
								$.log($.type(data));					// object
								
								data = $.stringify(data);
								$.log($.type(data));					// string
								
								data = $.parseJSON(data);
								$.log($.type(data));					// object

$.regexp

파라미터{String} 전달시 정규식을 반환한다.
parameters : tag, trim, number, date, rgb, engnum 
								$.regexp('tag');

$.strip

문자열에 있는 string 을 제거한다. 
								var domText = '<strong>Hello!</strong>';
								
								$.log($.strip(domText));		// Hello!

$.isNumberKey

key Event 발생시 Event 를 전달받아 숫자가 입력된 것인지 판별한다. 
								$('#input').on('keydown', function (e) {
									if(!$.isNumberKey(e)) return false;
								});

$.addCommas

전달받은 인자에 comma 를 추가하여 반환한다. 
								var price = 15000;
								
								var priceComma = $.addCommas(price);	// 15,000

$.preloadImage

이미지 Url 을 전달받아 사전에 로딩한다. 
								// 1개의 이미지를 로딩하는 경우.
								$.preloadImage('http://www.tru.co.kr/images/temp.jpg');
								
								// n개의 이미지를 로딩하는 경우.
								$.preloadImage([
									'http://www.tru.co.kr/images/temp0.jpg',
									'http://www.tru.co.kr/images/temp1.jpg',
									'http://www.tru.co.kr/images/temp2.jpg',
									'http://www.tru.co.kr/images/temp3.jpg'
								]);

$.loading

로딩바 Element 를 생성한다.
show, hide Method 를 지원한다. 
								// 로딩이미지 생성
								$.loading.show();
								
								// 생성시, 전달인자로 css 를 변경할 수 있다.
								$.loading.show({'z-index' : 200});
								
								// 로딩이미지 제거
								$.loading.hide();

$.mask

전체 화면을 Dim 하는 마스크를 생성한다.
show, hide Method 를 지원한다. 
								// 마스크 생성
								$.mask.show();
								
								// 생성시, 전달인자로 css 를 변경할 수 있다.
								$.mask.show({'z-index' : 200});
								
								// 마스크 제거
								$.mask.hide();
