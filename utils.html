$.config

설정된 application configuration 정보 get/set.

configuration 정보중 editable:true 인 값들만 runtime 에서 1회 설정이 가능하다.
설정된 정보를 get : readonly. 
								var _info = {
										'useLog' : true
									};
								
								// $.config > set
								$.config(_info);
							
								// $.config > get
								var root = $.config('root');

$.log

$.config('useLog') 설정이 true 일때 console 로그 작성. 
								$.log('Hello World!');

$.define

namespace 를 가둔 상태로 method 를 확장할 수 있다. 
								// define
								$.define('addMethod', function(){
									//...
								});
								
								// use
								$.addMethod();

$.cookie

기본적으로 cookie 는 browser, domain 기준으로 저장되며 string type 을 가진다.
일반적으로 4KB 까지만 저장이 가능합니다.
document.cookie 를 사용할 수 있는 get, set, remove, clear 을 지원한다. 
								// @method get
								// @params name
								$.cookie.get('name');
								
								// @method set
								// @params name
								// @params value
								// @params expireDays {Number}
								$.cookie.set('name', 1);
								
								// @method remove
								// @params name
								$.cookie.remove('name');
								
								// @method clear
								$.cookie.clear();

$.storage

supported IE 8+
localStorage 를 사용하기 위한 interface.
localStorage 는 일반적으로 4MB의 용량을 지원하며 String 으로 저장된다.
								// @method get
								// @params name
								$.storage.get('name');
								
								// @method set
								// @params name
								// @params value
								$.storage.set('name', 1);
								
								// @method remove
								// @params name
								$.storage.remove('name');
								
								// @method clear
								$.storage.clear();

$.stringify

JSON 객체를 지원하지 않는 IE 7 이하 브라우저에서 사용.
JSON.parse(), JSON.stringify() method 형태
$.parseJSON(), $.stringify() 로 대체 사용 가능. 
								var data = {"nation": "korea", "name": "jeros"};
								
								data = JSON.stringify(data);
								$.log($.type(data));					// string
								
								data = JSON.parse(data);
								$.log($.type(data));					// object
								
								data = $.stringify(data);
								$.log($.type(data));					// string
								
								data = $.parseJSON(data);
								$.log($.type(data));					// object

$.regexp

파라미터{String} 전달시 정규식을 반환한다.
parameters : tag, trim, number, date, rgb, engnum 
								$.regexp('tag');

$.strip

문자열에 있는 string 을 제거한다. 
								var domText = '<strong>Hello!</strong>';
								
								$.log($.strip(domText));		// Hello!

$.stateAlarm

instance message 를 띄울때 사용한다. 
								// @method stateAlarm
								// @params name
								// @params sec {Number}
								$.stateAlarm('message');

$.validate.init

validate plugin init : set defaults. 
								$.validate.init();
